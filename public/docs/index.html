GET api/v1/travels/{travel_slug}/tours

Example request:
curl --request GET \
    --get "http://localhost/api/v1/travels/nemo/tours" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"priceFrom\": 376041223.48716295,
    \"priceTo\": 1246363.5182438,
    \"dateFrom\": \"2024-02-13T20:27:13\",
    \"dateTo\": \"2024-02-13T20:27:13\",
    \"sortBy\": \"price\",
    \"sortOrder\": \"desc\"
}"

const url = new URL(
    "http://localhost/api/v1/travels/nemo/tours"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "priceFrom": 376041223.48716295,
    "priceTo": 1246363.5182438,
    "dateFrom": "2024-02-13T20:27:13",
    "dateTo": "2024-02-13T20:27:13",
    "sortBy": "price",
    "sortOrder": "desc"
};

fetch(url, {
    method: "GET",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
